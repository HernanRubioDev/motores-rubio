CREATE TABLE users ( 
	id_user SERIAL PRIMARY KEY, 
	name VARCHAR(40) NOT NULL, 
	surname VARCHAR(40) NOT NULL, 
	username VARCHAR(20) NOT NULL UNIQUE, 
	password VARCHAR(60) NOT NULL,
	auth_token VARCHAR(60) );

CREATE TABLE monophasics (
  motor_type VARCHAR(20) NOT NULL,
  id_monophasic SERIAL PRIMARY KEY,
  brand VARCHAR(20),
  rpm VARCHAR(20),
  minor_dim VARCHAR(20),
  model VARCHAR(20),
  hp VARCHAR(20),
  major_dim VARCHAR(20),
  owner VARCHAR(20),
  slots VARCHAR(20),
  large VARCHAR(20),
  motor_start VARCHAR(20),
  uf VARCHAR(20),
  work_steps VARCHAR(20),
  work_laps VARCHAR(20),
  work_wire VARCHAR(20),
  start_steps VARCHAR(20),
  start_laps VARCHAR(20),
  start_wire VARCHAR(20),
  voltage VARCHAR(20),
  connection VARCHAR(20),
  winding_type VARCHAR(20),
  observations VARCHAR(255),
  id_user integer REFERENCES users(id_user)
);

CREATE TABLE triphasics (
  motor_type VARCHAR(20) NOT NULL,
  id_triphasic SERIAL PRIMARY KEY,
  brand varchar(20),
  rpm varchar(20),
  minor_dim varchar(20),
  model varchar(20),
  hp varchar(20),
  major_dim varchar(20),
  owner varchar(20),
  slots varchar(20),
  large varchar(20),
  steps varchar(20),
  laps varchar(20),
  wire varchar(20),
  voltage varchar(20),
  connection varchar(20),
  winding_type varchar(20),
  observations varchar(255),
  id_user integer REFERENCES users(id_user)
);

CREATE TABLE washers (
  motor_type VARCHAR(20) NOT NULL,
  id_washer SERIAL PRIMARY KEY,
  brand VARCHAR(20),
  rpm VARCHAR(20),
  minor_dim VARCHAR(20),
  model VARCHAR(20),
  hp VARCHAR(20),
  major_dim VARCHAR(20),
  owner VARCHAR(20),
  slots VARCHAR(20),
  large VARCHAR(20),
  spin_work_steps VARCHAR(20),
  spin_work_laps VARCHAR(20),
  spin_work_wire VARCHAR(20),
  spin_start_steps VARCHAR(20),
  spin_start_laps VARCHAR(20),
  spin_start_wire VARCHAR(20),
  first_wash_work_steps VARCHAR(20),
  first_wash_work_laps VARCHAR(20),
  first_wash_work_wire VARCHAR(20),
  second_wash_work_steps VARCHAR(20),
  second_wash_work_laps VARCHAR(20),
  second_wash_work_wire VARCHAR(20),
  wash_start_steps VARCHAR(20),
  wash_start_laps VARCHAR(20),
  wash_start_wire VARCHAR(20),
  capacity VARCHAR(20),
  voltage VARCHAR(20),
  connection VARCHAR(20),
  winding_type VARCHAR(20),
  observations VARCHAR(255),
  id_user integer REFERENCES users(id_user)
);
